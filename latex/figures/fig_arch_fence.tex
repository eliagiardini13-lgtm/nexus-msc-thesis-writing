% Figure: the three-tier Fence safety classification.
\begin{tikzpicture}[nx,
  io/.style={draw,thick,rounded corners=3pt,align=center,
             font=\sffamily\footnotesize,minimum height=1.0cm},
  tier/.style={draw,thick,rounded corners=3pt,align=left,
               font=\sffamily\footnotesize,minimum width=8.7cm,
               minimum height=1.45cm,inner sep=6pt},
  badge/.style={draw,thick,rounded corners=3pt,align=center,
                font=\sffamily\footnotesize\bfseries,
                minimum width=2.1cm,minimum height=1.45cm},
  arr/.style={-{Latex[length=2.4mm]},thick},
]
  % ---- input + classifier ----
  \node[io,draw=nxGray,fill=white,minimum width=2.2cm] (con) at (-7.2,0) {Resolved\\contract};
  \node[io,draw=nxBlue,fill=nxBlueBg,minimum width=2.3cm,minimum height=4.6cm]
        (cls) at (-4.4,0) {Evaluate\\every\\safety\\rule};
  \draw[arr,nxGray] (con) -- (cls);

  % ---- three tiers ----
  \node[badge,fill=nxRedBg,draw=nxRed,text=nxRed]      (bcrit) at (-1.0, 2.05) {$\sigma_2$\\Critical};
  \node[badge,fill=nxYellowBg,draw=nxYellow,text=nxYellow](bwarn) at (-1.0, 0.0) {$\sigma_1$\\Warning};
  \node[badge,fill=nxGreenBg,draw=nxGreen,text=nxGreen] (bnorm) at (-1.0,-2.05) {$\sigma_0$\\Normal};

  \node[tier,fill=nxRedBg,draw=nxRed,anchor=west] (tcrit) at (0.15,2.05)
       {\textbf{Hard violation} of a safety rule (or negotiation failure).\\
        \emph{Now:} logged + flagged for audit. \quad
        \emph{Future:} runtime rejection / re-negotiation.};
  \node[tier,fill=nxYellowBg,draw=nxYellow,anchor=west] (twarn) at (0.15,0.0)
       {\textbf{Boundary approach} --- within 10\,\% of a safety limit.\\
        \emph{Now:} logged + flagged. \quad
        \emph{Future:} runtime alert.};
  \node[tier,fill=nxGreenBg,draw=nxGreen,anchor=west] (tnorm) at (0.15,-2.05)
       {\textbf{Safe} --- no rule violated, none at boundary.\\
        \emph{Now:} logged. \quad \emph{Future:} logged.};

  % ---- classifier to tiers ----
  \draw[arr,nxRed]    (cls.east|-bcrit) -- (bcrit.west);
  \draw[arr,nxYellow] (cls.east|-bwarn) -- (bwarn.west);
  \draw[arr,nxGreen]  (cls.east|-bnorm) -- (bnorm.west);

  % ---- log-only banner ----
  \node[font=\sffamily\footnotesize\itshape,nxGray] at (0.3,-3.35)
       {Fence is log-only within the scope of this thesis; active runtime enforcement is future work (Chapter 7).};
\end{tikzpicture}
