% Pandoc-friendly redefinitions of the custom macros from preamble.tex.
% This file is prepended to a chapter fragment before pandoc converts it
% to Word, so that custom commands expand into structures pandoc can
% render. It is NOT used by the LaTeX/PDF build (that uses preamble.tex).

% TikZ figures: the LaTeX build \input{}s native TikZ; for Word, pandoc
% includes a pre-rendered PNG (produced by scripts/render_figures.sh).
\newcommand{\nexustikzfigure}[3]{%
  \begin{figure}
  \centering
  \includegraphics[width=\linewidth]{figures/#3.png}
  \caption{#2}
  \label{#1}
  \end{figure}}

\newcommand{\nexusfigure}[3]{%
  \begin{figure}
  \centering
  \textit{[Figure placeholder --- #3]}
  \caption{#2}
  \label{#1}
  \end{figure}}

\newcommand{\PSA}{PSA}
\newcommand{\PMA}{PMA}
\newcommand{\QCA}{QCA}
\newcommand{\SCA}{SCA}
\newcommand{\NEXUS}{NEXUS}
